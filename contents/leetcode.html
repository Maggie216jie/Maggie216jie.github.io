Array functions

//common use functions
var addOne = function(x){
  return x+1;
};
var isEven = function (x) {
  return (x % 2 == 0) ? true : false;
};

var addTwo = function(x){
  x=x+2;
  console.log(x);
}; 
var numbers=[1,2,3,4,5,6];
var number2=[7,8];

// concat
numbers.concat(number2);
//result: [1,2,3,4,5,6,7,8]!!!but no change numbers and number2 

// every
numbers.every(addOne);
//result: true if all true return true, else false

// filter
numbers.filter(isEven);
//result:[2, 4, 6]

//forEach
numbers.forEach(addTwo);
//results: output: 3 4 5 6 7 8 9 no return result

//join
numbers.join('-')
//results: "1-2-3-4-5-6"

//indexOf
numbers.indexOf(1);
//results: 0 first index ,if not return -1

//lastIndexOf
numbers.lastIndexOf(1);
//results:0 largest index ,if not return -1

//map
numbers.map(addOne);
//results: [2, 3, 4, 5, 6, 7]

//reverse
numbers.reverse();
//results: [6, 5, 4, 3, 2, 1] numbers changed

//slice
numbers.slice(1,3);
//results: contains lower index, do not contains higher index

//some
numbers.some(isEven);
//results: false if all false return false, else true

//sort
numbers.sort();
//results: array change !!!zimuxun

//toString
numbers.toString();
//results: "1,2,3,4,5,6"
