Reducer

Code example of reducer

//trigger store refresh
function todoApp(state = initialState, action){
  switch (action.type) {
    case ADD_TODO:
      return Object.assign({}, state,{
      todos: [
        ...state.todos,
        {
          text: action.text,
          completed: false
        }
      ]
    })
    default:
      return state
   }
}
